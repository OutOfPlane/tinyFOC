cmake_minimum_required(VERSION 3.10)
project(tinyFOC)

# Set C standard
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Library from src directory
add_subdirectory(src)

# Tests executable
add_executable(tests_motorsim
    tests/motorsim/main.c
)
target_link_libraries(tests_motorsim PRIVATE tinyfoc)
